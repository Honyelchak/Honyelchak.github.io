int a[N], f[N];
int n;

int main()
{
    scanf("%d", &n);
    
    for (int i = 1; i <= n; i ++) scanf("%d", a + i);
    
    for (int i = 1; i <= n; i ++)
    {
        f[i] = 1;
        for (int j = 1; j < i; j ++)

                                          class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[][] f = new int[n][n + 1];

        for (int i = 0; i < n; i ++) {
            for (int j = i; j >= 0; j --) {
                if (nums[j] < nums[i]) {
                    if (f[i][0] < f[j][0] + 1) {
                        f[i] = Arrays.copyOf(f[j], n + 1);
                    }
                }
            }
            f[i][0] ++;
            f[i][f[i][0]] = nums[i];
        }
        int ans = 0;
        for (int i = 0; i < n; i ++) {
            ans = Math.max(ans, f[i][0]);
        }
            
        return ans;
    }
}
